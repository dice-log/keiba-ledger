
### 2-4. ローカルPostgreSQL スキーマ設計

```sql
-- ========================================
-- rawスキーマ：JRAvanの生データをそのまま保存
-- ========================================
CREATE SCHEMA IF NOT EXISTS raw;

-- 全データの入口。まずここに全部突っ込む
CREATE TABLE raw.jvdata (
    id          BIGSERIAL PRIMARY KEY,
    record_type TEXT NOT NULL,      -- 'RA', 'SE', 'O1', 'UM' etc.
    record_spec TEXT,               -- レコード種別仕様バージョン
    source_date DATE,               -- データの基準日
    payload     JSONB NOT NULL,     -- 生データをJSONで保存
    raw_text    TEXT,               -- 元の固定長テキストも保存
    fetched_at  TIMESTAMPTZ DEFAULT NOW(),
    processed   BOOLEAN DEFAULT FALSE  -- 正規化済みフラグ
);

CREATE INDEX idx_jvdata_type ON raw.jvdata(record_type);
CREATE INDEX idx_jvdata_date ON raw.jvdata(source_date);
CREATE INDEX idx_jvdata_processed ON raw.jvdata(processed);

-- 取得進捗管理（JV-Linkの差分管理補助）
CREATE TABLE raw.fetch_log (
    id              SERIAL PRIMARY KEY,
    fetch_type      TEXT,           -- 'initial', 'incremental'
    record_types    TEXT[],         -- 取得したレコード種別
    from_date       DATE,
    to_date         DATE,
    records_fetched INT,
    status          TEXT,           -- 'success', 'error', 'running'
    error_message   TEXT,
    started_at      TIMESTAMPTZ DEFAULT NOW(),
    finished_at     TIMESTAMPTZ
);

-- ========================================
-- analyticsスキーマ：正規化済み・分析用
-- ========================================
CREATE SCHEMA IF NOT EXISTS analytics;

-- レース基本情報
CREATE TABLE analytics.races (
    race_id         TEXT PRIMARY KEY,   -- JRAレースID（例: 2024010108010101）
    race_date       DATE NOT NULL,
    venue_code      TEXT NOT NULL,      -- 開催場コード
    venue_name      TEXT NOT NULL,      -- '東京' '阪神' etc.
    race_number     INT,
    race_name       TEXT,
    grade           TEXT,               -- 'G1' 'G2' 'G3' 'OP' etc.
    surface         TEXT,               -- '芝' 'ダート' '障害'
    distance        INT,                -- メートル
    direction       TEXT,               -- '右' '左' '直線'
    weather         TEXT,               -- '晴' '曇' '雨' '小雨' '雪'
    track_condition TEXT,               -- '良' '稍重' '重' '不良'
    field_count     INT,                -- 出走頭数
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_races_date ON analytics.races(race_date DESC);
CREATE INDEX idx_races_grade ON analytics.races(grade);
CREATE INDEX idx_races_venue ON analytics.races(venue_name);

-- 出走馬・着順
CREATE TABLE analytics.race_entries (
    id              BIGSERIAL PRIMARY KEY,
    race_id         TEXT REFERENCES analytics.races(race_id),
    horse_id        TEXT NOT NULL,
    horse_name      TEXT NOT NULL,
    horse_number    INT,                -- 馬番
    frame_number    INT,                -- 枠番
    jockey_id       TEXT,
    jockey_name     TEXT,
    trainer_id      TEXT,
    trainer_name    TEXT,
    weight_carried  NUMERIC(4,1),       -- 斤量
    horse_weight    INT,                -- 馬体重
    weight_diff     INT,                -- 馬体重増減
    finish_pos      INT,                -- 着順（NULL=除外・取消）
    finish_time     NUMERIC(6,1),       -- タイム（秒）
    last_3f         NUMERIC(4,1),       -- 上がり3F（秒）
    win_odds        NUMERIC(7,1),       -- 単勝オッズ
    place_odds_min  NUMERIC(6,1),       -- 複勝オッズ下限
    place_odds_max  NUMERIC(6,1),       -- 複勝オッズ上限
    popularity      INT,                -- 人気順
    running_style   TEXT,               -- '逃げ' '先行' '差し' '追込'
    corner_pos      TEXT,               -- コーナー通過順位 '3-3-4-5'
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_entries_race ON analytics.race_entries(race_id);
CREATE INDEX idx_entries_horse ON analytics.race_entries(horse_id);
CREATE INDEX idx_entries_jockey ON analytics.race_entries(jockey_id);

-- 競走馬マスタ
CREATE TABLE analytics.horses (
    horse_id        TEXT PRIMARY KEY,
    name            TEXT NOT NULL,
    name_kana       TEXT,
    birth_date      DATE,
    sex             TEXT,               -- '牡' '牝' 'セ'
    coat_color      TEXT,               -- '鹿毛' '黒鹿毛' etc.
    sire_id         TEXT,               -- 父
    sire_name       TEXT,
    dam_id          TEXT,               -- 母
    dam_name        TEXT,
    broodmare_sire  TEXT,               -- 母父
    trainer_id      TEXT,
    owner_name      TEXT,
    breeder_name    TEXT,
    created_at      TIMESTAMPTZ DEFAULT NOW(),
    updated_at      TIMESTAMPTZ DEFAULT NOW()
);

-- 騎手マスタ
CREATE TABLE analytics.jockeys (
    jockey_id       TEXT PRIMARY KEY,
    name            TEXT NOT NULL,
    name_kana       TEXT,
    belong_to       TEXT,               -- 所属（美浦・栗東等）
    birth_date      DATE,
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

-- オッズ（最終オッズ）
CREATE TABLE analytics.odds_final (
    id              BIGSERIAL PRIMARY KEY,
    race_id         TEXT REFERENCES analytics.races(race_id),
    bet_type        TEXT NOT NULL,      -- 'WIN' 'PLACE' 'QUINELLA' etc.
    combination     TEXT NOT NULL,      -- '1' '1-3' '1-3-5' etc.
    odds            NUMERIC(8,1),
    popularity      INT,
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_odds_race ON analytics.odds_final(race_id);

-- オッズ時系列（単複・馬連・枠連 過去1年分）
CREATE TABLE analytics.odds_timeseries (
    id              BIGSERIAL PRIMARY KEY,
    race_id         TEXT REFERENCES analytics.races(race_id),
    bet_type        TEXT NOT NULL,
    combination     TEXT NOT NULL,
    odds            NUMERIC(8,1),
    recorded_at     TIMESTAMPTZ NOT NULL,   -- 記録時刻
    minutes_to_start INT                    -- 発走まで何分前
);

CREATE INDEX idx_odds_ts_race ON analytics.odds_timeseries(race_id);
CREATE INDEX idx_odds_ts_time ON analytics.odds_timeseries(recorded_at);

-- 払戻
CREATE TABLE analytics.payouts (
    id              BIGSERIAL PRIMARY KEY,
    race_id         TEXT REFERENCES analytics.races(race_id),
    bet_type        TEXT NOT NULL,
    combination     TEXT NOT NULL,
    payout          INT NOT NULL,       -- 払戻金額（100円あたり）
    popularity      INT,
    created_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_payouts_race ON analytics.payouts(race_id);
```

### 2-5. Pythonスクリプト仕様

#### jvlink_client.py — JV-Linkラッパー

```python
"""
JV-Linkをwin32comでラップしてデータ取得を抽象化する。
JV-LinkはWindows専用COMオブジェクト。

主要メソッド:
- fetch_stored_data(record_types, from_date, to_date) -> List[dict]
  蓄積系データの取得（過去データ）
- fetch_realtime_data(record_types) -> List[dict]
  速報系データの取得（当日オッズ等）
- get_last_file_timestamp() -> str
  前回取得の続きから取得するためのタイムスタンプ取得

依存: pywin32, python-dotenv
"""
```

#### initial_fetch.py — 初回全データ取得

```python
"""
初回1回だけ実行する全データ取得スクリプト。
実行時間の目安: オッズなし5〜10時間、オッズあり10〜15時間

取得対象レコード種別:
- RA: レース詳細
- SE: 馬毎レース情報（着順・タイム）
- HR: 払戻
- O1: オッズ（単複）
- O3: オッズ（馬連）
- UM: 競走馬マスタ
- KS: 騎手マスタ
- CH: 調教師マスタ
- WH: 馬体重
- WE: 天候・馬場

取得範囲: 2010年1月1日〜現在（過去15年で十分）
保存先: raw.jvdataテーブル（JSONB）

実行方法:
  python scripts/fetch/initial_fetch.py
  python scripts/fetch/initial_fetch.py --from 2015-01-01  # 開始日指定
  python scripts/fetch/initial_fetch.py --no-odds          # オッズ除外
"""
```

#### incremental_fetch.py — 差分取得（定期実行）

```python
"""
差分データのみ取得して追記するスクリプト。
Windowsタスクスケジューラから毎週月曜朝6時に自動実行。

処理の流れ:
1. raw.fetch_logから前回成功時刻を取得
2. JV-LinkにLastFileTimestampを問い合わせ
3. 差分データをraw.jvdataに追記
4. normalize.pyを呼び出してanalyticsスキーマに反映
5. sync_to_supabase.pyを呼び出して直近データをSupabaseに同期
6. ログ記録

実行方法:
  python scripts/fetch/incremental_fetch.py
"""
```

#### normalize.py — rawデータ→analyticsテーブル変換

```python
"""
raw.jvdata（JSONB）をパースしてanalyticsスキーマの各テーブルに変換・挿入。
processed=FALSEのレコードのみ処理し、完了後processed=TRUEに更新。

処理対象:
- RA → analytics.races
- SE → analytics.race_entries
- HR → analytics.payouts
- O1/O3 → analytics.odds_final, analytics.odds_timeseries
- UM → analytics.horses
- KS → analytics.jockeys
"""
```

#### sync_to_supabase.py — Supabase同期

```python
"""
ローカルPostgreSQLの直近データのみSupabaseに同期する。
全データをSupabaseに上げるとコスト高のため、アプリで必要な分だけ送る。

同期対象:
- analytics.races: 直近2年分
- analytics.race_entries: 直近2年分
- analytics.horses: 全件（軽量）
- analytics.jockeys: 全件（軽量）
- analytics.odds_final: 直近1年分
- analytics.payouts: 直近2年分
- prediction.scores: 全件（ML予測結果 Phase 3で追加）

同期方式: UPSERT（race_idをキーに重複防止）
"""
```

```sql
CREATE TABLE bets (
```

## 6. 重要な制約・注意事項

```
【JRAvan利用規約】
- 取得データの私的利用は問題なし
- データそのままの公開・販売はNG
- 独自に加工した指数・予測結果の提供はグレーゾーン
  → アプリでは「独自スコア」として表示し生データは非表示

【技術的制約】
- JV-LinkはWindows専用（Mac不可）
- データ取得はWindowsローカルで実行必須
- Supabaseへの同期はローカルから手動 or タスクスケジューラ

【ML制約】
- 過学習注意：学習期間とテスト期間を厳密に分ける
- オッズは締切直前のものを使う（早い段階のオッズは予測に使わない）
- バックテストで手数料（JRAは約25%）を必ず考慮する
```
